class {{ClassName}}ListScreen extends StatefulWidget {
  const {{ClassName}}ListScreen({super.key});
  static String keyName = '/{{class_name}}_list';

  @override
  State<{{ClassName}}ListScreen> createState() => _{{ClassName}}ListScreenState();
}

class _{{ClassName}}ListScreenState extends State<{{ClassName}}ListScreen> {
  // Repository và Cubit (inject từ DI)
  final {{ClassName}}Repository _repository = di<{{ClassName}}Repository>();
  final {{ClassName}}Cubit _cubit = di<{{ClassName}}Cubit>();
  int _currentTabIndex = 0;
  String? _currentKeyword;

  @override
  Widget build(BuildContext context) {
    return ListScreenTemplate<{{ClassName}}>(
      // Required parameters
      appBarTitle: 'Danh sách {{DisplayName}}',
      searchHint: 'Tìm kiếm theo tên, code...',
      
      // Tabs configuration
      tabs: [
        TabConfig(label: 'Tất cả'),
        TabConfig(
          label: 'Mới',
          // badgeStream: _repository.new{{ClassName}}sCounter.stream, // Optional - uncomment nếu có
          // hasPermission: () => PermissionHelper.view(AbilitySubject.{{class_name}}), // Optional
        ),
        TabConfig(label: 'Đang xử lý'),
      ],
      initialTabIndex: _currentTabIndex,
      onTabChanged: (index) {
        setState(() {
          _currentTabIndex = index;
        });
        // Reload data cho tab mới
        _cubit.loadDataForTab(
          tabIndex: index,
          keyword: _currentKeyword,
        );
      },
      
      // List configuration
      dataStream: _repository.{{class_name}}sStream,
      totalCount: _repository.total{{ClassName}}s,
      itemBuilder: (context, item, index) {
        return Container(
          margin: EdgeInsets.symmetric(horizontal: 16.w),
          padding: EdgeInsets.all(10.w),
          decoration: BoxDecoration(
            color: AppColors.white,
            borderRadius: BorderRadius.circular(10.r),
            border: Border.all(color: AppColors.greyC0),
          ),
          child: _{{ClassName}}ItemCard(item: item),
        );
      },
      emptyMessage: 'Không tìm thấy {{DisplayName}}',
      
      // Callbacks (REQUIRED)
      onSearch: (keyword) {
        _currentKeyword = keyword;
        _cubit.filter{{ClassName}}s(keyword: keyword);
      },
      onRefresh: () async {
        await _cubit.loadDataForTab(
          tabIndex: _currentTabIndex,
          offset: 0,
          keyword: _currentKeyword,
        );
      },
      onLoadMore: () {
        final current = _repository.{{class_name}}s;
        if (current.length >= _repository.total{{ClassName}}s) return true;
        _cubit.loadMore{{ClassName}}s(
          tabIndex: _currentTabIndex,
          offset: current.length,
          keyword: _currentKeyword,
        );
        return true;
      },
    );
  }
}
